SELECT * FROM EMP e, DEPT d WHERE e.DEPTNO = d.DEPTNO;

SELECT e.EMPNO, e.ENAME, e.SAL, d.DEPTNO FROM EMP e, DEPT d WHERE e.DEPTNO = d.DEPTNO;

SELECT e.EMPNO, e.ENAME, e.SAL, d.DEPTNO FROM EMP e, DEPT d WHERE e.DEPTNO = d.DEPTNO AND e.ENAME LIKE 'S%';

SELECT e.EMPNO, e.ENAME, e.SAL, d.DEPTNO FROM EMP e INNER JOIN DEPT d ON e.DEPTNO = d.DEPTNO;

SELECT e.EMPNO, e.ENAME, e.SAL, d.DEPTNO FROM EMP e INNER JOIN DEPT d ON e.DEPTNO = d.DEPTNO AND e.ENAME LIKE 'S%';

SELECT DEPTNO FROM EMP INTERSECT SELECT DEPTNO FROM DEPT;

SELECT * FROM EMP e LEFT OUTER JOIN DEPT d ON e.DEPTNO = d.DEPTNO;

SELECT * FROM EMP e RIGHT OUTER JOIN DEPT d ON e.DEPTNO = d.DEPTNO;

SELECT COUNT(*) FROM EMP CROSS JOIN DEPT;

SELECT COUNT(*) FROM EMP, DEPT;

SELECT COUNT(*) FROM (SELECT * FROM EMP UNION SELECT * FROM EMP);

SELECT COUNT(*) FROM (SELECT * FROM EMP UNION ALL SELECT * FROM EMP);

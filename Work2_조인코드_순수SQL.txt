SELECT * FROM EMP, DEPT
WHERE EMP.DEPTNO = DEPT.DEPTNO;

SELECT * FROM EMP e, DEPT d
WHERE e.DEPTNO = d.DEPTNO;

SELECT e.EMPNO, e.ENAME, e.SAL, d.DEPTNO
FROM EMP e, DEPT d
WHERE e.DEPTNO = d.DEPTNO;

SELECT e.EMPNO, e.ENAME, e.SAL, d.DEPTNO
FROM EMP e, DEPT d
WHERE e.DEPTNO = d.DEPTNO
  AND e.ENAME LIKE '김%';

SELECT e.EMPNO, e.ENAME, e.SAL, d.DEPTNO
FROM EMP e
INNER JOIN DEPT d
  ON e.DEPTNO = d.DEPTNO;

SELECT e.EMPNO, e.ENAME, e.SAL, d.DEPTNO
FROM EMP e
INNER JOIN DEPT d
  ON e.DEPTNO = d.DEPTNO
 AND e.ENAME LIKE '김%';

SELECT DEPTNO FROM EMP
INTERSECT
SELECT DEPTNO FROM DEPT;

SELECT * FROM EMP e
LEFT OUTER JOIN DEPT d
  ON e.DEPTNO = d.DEPTNO;

SELECT * FROM EMP e
RIGHT OUTER JOIN DEPT d
  ON e.DEPTNO = d.DEPTNO;

SELECT * FROM EMP
CROSS JOIN DEPT;

SELECT * FROM EMP, DEPT;

SELECT * FROM EMP
UNION
SELECT * FROM EMP;

SELECT * FROM EMP
UNION ALL
SELECT * FROM EMP;

SELECT DEPTNO FROM DEPT
MINUS
SELECT DEPTNO FROM EMP;

SELECT MAX(LEVEL)
FROM NEXTEMP
START WITH MGR IS NULL
CONNECT BY PRIOR EMPNO = MGR;

SELECT LEVEL, EMPNO, MGR, ENAME
FROM NEXTEMP
START WITH MGR IS NULL
CONNECT BY PRIOR EMPNO = MGR;

SELECT LEVEL,
       LPAD(' ', 4*(LEVEL-1)) || EMPNO AS EMPNO,
       MGR,
       CONNECT_BY_ISLEAF
FROM NEXTEMP
START WITH MGR IS NULL
CONNECT BY PRIOR EMPNO = MGR;

SELECT LEVEL,
       LPAD(' ', 4*(LEVEL-1)) || EMPNO AS EMPNO,
       MGR,
       CONNECT_BY_ISLEAF
FROM NEXTEMP
START WITH EMPNO = 1008
CONNECT BY PRIOR MGR = EMPNO;
